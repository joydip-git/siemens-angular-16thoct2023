interface IOperations{
	add(a:number, b:number):number;
}
class Operations implements IOperations{
	add(a:number, b:number):number{
		return a+b
	}
}

interface IPerson{
	name:string;
	id:number;
	salary?:number;
}

class Person implements IPerson{
	name:string;
	id:number;
	salary:number;
	constructor(name:string, id:number, salary:number){
		this.name = name
		this.id = id
		this.salary = salary
	}
}
